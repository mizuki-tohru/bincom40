#安定化電源 ALIENTEK DP100設定
NAME "DP100"

#デバイスアドレスを初期値の1と仮定
INIT START
INIT "INFO"
#INIT "ACT"
INIT END

CMD NAME "SET0V"
CMD "VOLT 0.0"
CMD "STOP"
CMD END

CMD NAME "SETV"
CMD "VOLT" P1
CMD END

CMD NAME "SETV1M"
CMD "VOLT" P1
CMD MWAIT 1
CMD "STOP"
CMD END

CMD NAME "ACT"
CMD "ACT"
CMD END

CMD NAME "READ"
CMD "READ"
CMD END
